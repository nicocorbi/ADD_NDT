package com.example.practica12_13_14;

import org.springframework.data.annotation.Id;
import org.springframework.data.mongodb.core.mapping.Document;
import java.util.Date;

@Document(collection = "AccesoADatosCollection")
public class Transaction {

    @Id
    private String id;
    private String description;
    private double quantity;
    private boolean income;
    private Date date;

    // Constructor con parámetros
    public Transaction(String description, double quantity, boolean income, Date date) {
        this.description = description;
        this.quantity   = quantity;
        this.income      = income;
        this.date        = date;
    }

    // Getters
    public String  getId()          { return id; }
    public String  getDescription() { return description; }
    public double  getQuantity()    { return quantity; }
    public boolean isIncome()       { return income; }
    public Date    getDate()        { return date; }

    // Setters
    public void setId         (String  id)          { this.id          = id; }
    public void setDescription(String  description) { this.description = description; }
    public void setQuantity   (double  quantity)    { this.quantity    = quantity; }
    public void setIncome     (boolean income)      { this.income      = income; }
    public void setDate       (Date    date)        { this.date        = date; }

    @Override
    public String toString() {
        return "Transaction{" +
               "id='"          + id          + "'" +
               ", description='" + description + "'" +
               ", quantity="   + quantity    +
               ", income="     + income      +
               ", date="       + date        +
               '}';
    }
}
PARTE 9 – Repositorio

TransactionRepository.java

src/main/java/com/example/practica12_13_14/TransactionRepository.java

package com.example.practica12_13_14;

import org.springframework.data.mongodb.repository.MongoRepository;
import org.springframework.data.mongodb.repository.Query;
import java.util.Date;
import java.util.List;

public interface TransactionRepository extends MongoRepository<Transaction, String> {

    // Obtener los gastos (income = false)
    List<Transaction> findByIncomeFalse();

    // Obtener los ingresos (income = true)
    List<Transaction> findByIncomeTrue();

    // Transacciones del último mes (dateAfter)
    List<Transaction> findByDateAfter(Date date);

    // Transacciones con valor superior a un valor
    List<Transaction> findByQuantityGreaterThan(double quantity);

    // Transacciones con valor entre dos valores
    List<Transaction> findByQuantityBetween(double min, double max);

    // Gastos entre dos fechas
    List<Transaction> findByIncomeFalseAndDateBetween(Date start, Date end);

    // Gastos con cantidad menor a un valor dado
    List<Transaction> findByIncomeFalseAndQuantityLessThan(double quantity);

    // Gastos con cantidad mayor a un valor dado
    List<Transaction> findByIncomeFalseAndQuantityGreaterThan(double quantity);

    // Buscar por descripción que contenga texto (para la parte 10)
    List<Transaction> findByDescriptionContaining(String text);
}

package com.example.practica12_13_14;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.CommandLineRunner;
import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

import java.util.Calendar;
import java.util.Date;
import java.util.List;

@SpringBootApplication
public class Practica121314Application implements CommandLineRunner {

    @Autowired
    private TransactionRepository transactionRepository;

    public static void main(String[] args) {
        SpringApplication.run(Practica121314Application.class, args);
    }

    @Override
    public void run(String... args) throws Exception {

        // ── 1. Borrar todos los datos ─────────────────────────────────────
        transactionRepository.deleteAll();
        System.out.println("=== Datos borrados ===");

        // ── 2. Insertar datos de prueba ───────────────────────────────────
        // Helper para crear fechas fácilmente
        Calendar cal = Calendar.getInstance();

        cal.set(2024, Calendar.JANUARY, 10);
        Date fecha1 = cal.getTime();

        cal.set(2024, Calendar.FEBRUARY, 5);
        Date fecha2 = cal.getTime();

        cal.set(2024, Calendar.MARCH, 20);
        Date fecha3 = cal.getTime();

        // Fecha del mes actual (para probar últimoMes)
        cal.setTime(new Date());
        cal.add(Calendar.DAY_OF_MONTH, -5);
        Date fechaReciente = cal.getTime();

        transactionRepository.save(new Transaction("Salario enero",       1500.00, true,  fecha1));
        transactionRepository.save(new Transaction("Compra supermercado", 80.50,  false, fecha1));
        transactionRepository.save(new Transaction("Freelance web",       350.00, true,  fecha2));
        transactionRepository.save(new Transaction("Compra ropa",         120.00, false, fecha2));
        transactionRepository.save(new Transaction("Factura electricidad", 55.30, false, fecha3));
        transactionRepository.save(new Transaction("Compra online Amazon", 200.00, false, fecha3));
        transactionRepository.save(new Transaction("Ingreso alquiler",    600.00, true,  fechaReciente));

        System.out.println("=== Datos de prueba insertados ===");

        // ── 3. Consultas ─────────────────────────────────────────────────

        // a) Todos los ingresos
        System.out.println("\n--- INGRESOS ---");
        List<Transaction> ingresos = transactionRepository.findByIncomeTrue();
        ingresos.forEach(System.out::println);

        // b) Todos los gastos
        System.out.println("\n--- GASTOS ---");
        List<Transaction> gastos = transactionRepository.findByIncomeFalse();
        gastos.forEach(System.out::println);

        // c) Transacciones entre dos fechas
        System.out.println("\n--- TRANSACCIONES ENTRE ENE y FEB 2024 ---");
        cal.set(2024, Calendar.JANUARY, 1);
        Date inicio = cal.getTime();
        cal.set(2024, Calendar.FEBRUARY, 28);
        Date fin = cal.getTime();
        List<Transaction> entresFechas = transactionRepository.findByIncomeFalseAndDateBetween(inicio, fin);
        entresFechas.forEach(System.out::println);

        // d) Transacciones con "Compra" en la descripción
        System.out.println("\n--- TRANSACCIONES CON 'Compra' EN DESCRIPCIÓN ---");
        List<Transaction> compras = transactionRepository.findByDescriptionContaining("Compra");
        compras.forEach(System.out::println);

        // e) Transacciones con cantidad mayor a 100
        System.out.println("\n--- TRANSACCIONES CON CANTIDAD > 100 ---");
        List<Transaction> mayores = transactionRepository.findByQuantityGreaterThan(100.0);
        mayores.forEach(System.out::println);
    }
}
